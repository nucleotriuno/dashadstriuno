# Alexa Semijoias — Landing Page Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a single-file static landing page (`index.html`) that captures consultant leads for Alexa Semijoias via a form that POSTs JSON to an n8n webhook.

**Architecture:** Everything lives in one `index.html` file — inline CSS via `<style>`, inline JS via `<script>`, all SVGs inline. The page is 11 sections in a single-scroll layout. JavaScript handles three things: sticky navbar active state, FAQ accordion, and form submission via `fetch()`. No build step, no dependencies, no frameworks.

**Tech Stack:** HTML5, CSS3 (custom properties, flexbox, grid), Vanilla JavaScript (ES6+), Google Fonts (Cormorant + Montserrat)

**Spec:** `docs/superpowers/specs/2026-03-23-alexa-lp-design.md`

## CSS Architecture (inside `<style>` in `<head>`)

```
:root                   → 4 color tokens + font tokens
html                    → scroll-behavior: smooth
body                    → font-family Montserrat, background base
.navbar                 → sticky top-0, z-100, backdrop-filter
.hero                   → gradient bg + photo bg setup
.section-*              → each section's background + padding
.benefits-grid          → CSS grid, responsive columns
.steps-grid             → CSS grid 2×2
.form-split             → flexbox split layout
.accordion-item         → FAQ accordion base styles
@media (≥768px)         → tablet overrides
@media (≥1024px)        → desktop overrides
```

```javascript
const WEBHOOK_URL = 'SUBSTITUIR_PELA_URL_DO_WEBHOOK_N8N';

// ── FAQ Accordion ──
document.querySelectorAll('.faq-item').forEach(item => {
  const trigger = item.querySelector('.faq-item__trigger');
  const body = item.querySelector('.faq-item__body');

  trigger.addEventListener('click', () => {
    const isOpen = item.dataset.open === 'true';

    // Close all other items
    document.querySelectorAll('.faq-item').forEach(other => {
      if (other !== item) {
        other.dataset.open = 'false';
        other.querySelector('.faq-item__trigger').setAttribute('aria-expanded', 'false');
        other.querySelector('.faq-item__body').style.maxHeight = '0';
      }
    });

    // Toggle current
    if (isOpen) {
      item.dataset.open = 'false';
      trigger.setAttribute('aria-expanded', 'false');
      body.style.maxHeight = '0';
    } else {
      item.dataset.open = 'true';
      trigger.setAttribute('aria-expanded', 'true');
      body.style.maxHeight = body.scrollHeight + 'px';
    }
  });
});
```

```javascript
// ── Form submission ──
const form = document.getElementById('lead-form');
const submitBtn = document.getElementById('submit-btn');
const formSuccess = document.getElementById('form-success');
const formError = document.getElementById('form-error');

function showFieldError(fieldId, message) {
  const input = document.getElementById(fieldId);
  const error = document.getElementById(fieldId + '-error');
  if (input) input.setAttribute('aria-invalid', 'true');
  if (error) error.textContent = message;
}

function clearFieldError(fieldId) {
  const input = document.getElementById(fieldId);
  const error = document.getElementById(fieldId + '-error');
  if (input) input.removeAttribute('aria-invalid');
  if (error) error.textContent = '';
}

function clearAllErrors() {
  ['nome', 'whatsapp', 'email', 'cidade'].forEach(clearFieldError);
  formError.textContent = '';
}

function validateForm(data) {
  let valid = true;

  if (!data.nome || data.nome.trim().length < 3) {
    showFieldError('nome', 'Por favor, informe seu nome completo.');
    valid = false;
  }

  if (!data.whatsapp || data.whatsapp.replace(/\D/g, '').length < 10) {
    showFieldError('whatsapp', 'Por favor, informe um WhatsApp válido.');
    valid = false;
  }

  const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  if (!data.email || !emailRegex.test(data.email)) {
    showFieldError('email', 'Por favor, informe um e-mail válido.');
    valid = false;
  }

  if (!data.cidade || data.cidade.trim().length < 2) {
    showFieldError('cidade', 'Por favor, informe sua cidade.');
    valid = false;
  }

  if (!data.consentimento) {
    formError.textContent = 'Por favor, aceite os termos para continuar.';
    valid = false;
  }

  return valid;
}

// Clear error on input
['nome', 'whatsapp', 'email', 'cidade'].forEach(fieldId => {
  const input = document.getElementById(fieldId);
  if (input) {
    input.addEventListener('blur', () => {
      if (input.value.trim()) clearFieldError(fieldId);
    });
  }
});

form.addEventListener('submit', async (e) => {
  e.preventDefault();
  clearAllErrors();

  const data = {
    nome: document.getElementById('nome').value.trim(),
    whatsapp: document.getElementById('whatsapp').value.trim(),
    email: document.getElementById('email').value.trim(),
    cidade: document.getElementById('cidade').value.trim(),
    consentimento: document.getElementById('consentimento').checked,
    origem: 'landing-page-consultoras',
    timestamp: new Date().toISOString(),
  };

  if (!validateForm(data)) {
    // Focus first invalid field
    const firstInvalid = form.querySelector('[aria-invalid="true"]');
    if (firstInvalid) firstInvalid.focus();
    return;
  }

  // Loading state
  submitBtn.disabled = true;
  submitBtn.textContent = 'Enviando...';

  try {
    const response = await fetch(WEBHOOK_URL, {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(data),
    });

    if (!response.ok) throw new Error('Resposta inesperada do servidor.');

    // Success
    form.hidden = true;
    formSuccess.hidden = false;
    // Move focus to success message so screen readers announce it
    formSuccess.setAttribute('tabindex', '-1');
    formSuccess.focus();
    formSuccess.scrollIntoView({ behavior: 'smooth', block: 'center' });

  } catch (err) {
    formError.textContent = 'Não foi possível enviar. Por favor, tente novamente.';
    submitBtn.disabled = false;
    submitBtn.textContent = 'Quero me inscrever';
  }
});
```
